var obj = {
    handler : null,
    myhandler : function () {
        console.log('click');
        this.target.removeEventListener('click', this.handler);
    },
    target : document.getElementById('BtnID'),
    init : function () {
        this.handler = this.myhandler.bind(this);
        this.target.addEventListener('click', this.handler);
    },
};
obj.init();
